int ans=0;
void f(node,adj,wt,val){
    ans+=abs(wt[node]-val);
    for(int child:adj[node]){
        f(child,adj,wt,val+wt[node]-val);
    }
    return;
}

int main(){
    int n;
    cin>>n;
    vector<int>wt(n);
    for(int i=0;i<n;i++){
        cin>>wt[i];
        cin>>parent[i];
        if(parent[i]==-1) root=i;
        else{
            adj[parent[i]].push_back(i);
        }
    }

    f(root,adj,wt,0);
}
