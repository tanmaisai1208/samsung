accs, maes, rmses = [], [], []
    for d in day_data:
        model.fit(d["X_train"], d["y_train"])
        y_pred   = model.predict(d["X_pred"])[0]
        y_actual = d["y_actual"]

        # Only evaluate on hours where ground truth is not NaN
        valid_mask = ~np.isnan(y_actual)
        if valid_mask.sum() == 0:
            continue   # no valid hours to evaluate on this day

        y_actual_valid = y_actual[valid_mask]
        y_pred_valid   = y_pred[valid_mask]

        accs.append(accuracy_threshold(y_actual_valid, y_pred_valid))
        maes.append(mean_absolute_error(y_actual_valid, y_pred_valid))
        rmses.append(np.sqrt(mean_squared_error(y_actual_valid, y_pred_valid)))
